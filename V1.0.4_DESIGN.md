```typescript
// services/monitor/types.ts

/**
 * 告警级别
 */
export type AlertSeverity = 'info' | 'warning' | 'error' | 'critical';

/**
 * 告警类型
 */
export type AlertType = 
  | 'consumer_lag'      // 消费延迟
  | 'connection_lost'   // 连接断开
  | 'topic_growth'      // Topic 增长
  | 'produce_error'     // 生产错误
  | 'consume_error';    // 消费错误

/**
 * 告警规则
 */
export interface AlertRule {
  id: string;
  name: string;
  enabled: boolean;
  type: AlertType;
  
  // 触发条件
  condition: {
    metric: string;
    operator: 'gt' | 'gte' | 'lt' | 'lte' | 'eq';
    threshold: number;
    duration: number;        // 持续多少秒触发（防抖）
    cooldown: number;        // 冷却时间（秒）
  };
  
  // 通知配置
  notification: {
    desktop: boolean;
    sound: boolean;
    badge: boolean;
    autoDismiss: number;     // 自动消失时间（秒），0表示不自动消失
  };
}

/**
 * 告警实例
 */
export interface Alert {
  id: string;
  ruleId: string;
  type: AlertType;
  severity: AlertSeverity;
  title: string;
  message: string;
  metadata: AlertMetadata;
  
  // 状态
  status: 'active' | 'acknowledged' | 'resolved';
  createdAt: number;
  acknowledgedAt?: number;
  resolvedAt?: number;
  
  // 通知状态
  notified: {
    desktop: boolean;
    sound: boolean;
  };
}

/**
 * 告警元数据 - 根据类型不同而变化
 */
export type AlertMetadata = 
  | ConsumerLagMetadata
  | ConnectionLostMetadata
  | TopicGrowthMetadata
  | ErrorMetadata;

export interface ConsumerLagMetadata {
  connectionId: string;
  consumerGroup: string;
  topic: string;
  partition: number;
  currentLag: number;
  threshold: number;
}

export interface ConnectionLostMetadata {
  connectionId: string;
  connectionName: string;
  error?: string;
  retryCount: number;
  maxRetries: number;
}

export interface TopicGrowthMetadata {
  connectionId: string;
  topic: string;
  messagesPerMinute: number;
  threshold: number;
  totalMessages: number;
}

export interface ErrorMetadata {
  connectionId: string;
  operation: string;
  errorCode: string;
  errorMessage: string;
  retryable: boolean;
}

/**
 * 告警统计
 */
export interface AlertStats {
  total: number;
  active: number;
  warning: number;
  error: number;
  critical: number;
}
```

### 1.3 核心服务实现

```typescript
// services/monitor/MonitorService.ts

import { EventEmitter } from 'events';

export class MonitorService extends EventEmitter {
  private static instance: MonitorService;
  private rules: AlertRule[] = [];
  private alerts: Alert[] = [];
  private timers: Map<string, NodeJS.Timeout> = new Map();
  private lagCache: Map<string, number> = new Map();
  private lastAlertTime: Map<string, number> = new Map();
  
  private readonly MAX_ALERTS = 100;
  private readonly STORAGE_KEY = 'kafkit-alerts';
  private readonly RULES_KEY = 'kafkit-alert-rules';
  
  private constructor() {
    super();
    this.loadFromStorage();
    this.startMonitoring();
  }
  
  static getInstance(): MonitorService {
    if (!MonitorService.instance) {
      MonitorService.instance = new MonitorService();
    }
    return MonitorService.instance;
  }
  
  // ============ 规则管理 ============
  
  getRules(): AlertRule[] {
    return [...this.rules];
  }
  
  addRule(rule: Omit<AlertRule, 'id'>): AlertRule {
    const newRule: AlertRule = {
      ...rule,
      id: crypto.randomUUID()
    };
    this.rules.push(newRule);
    this.saveRules();
    this.startRuleMonitoring(newRule);
    return newRule;
  }
  
  updateRule(id: string, updates: Partial<AlertRule>): AlertRule | null {
    const index = this.rules.findIndex(r => r.id === id);
    if (index === -1) return null;
    
    this.rules[index] = { ...this.rules[index], ...updates };
    this.saveRules();
    this.restartRuleMonitoring(this.rules[index]);
    return this.rules[index];
  }
  
  deleteRule(id: string): boolean {
    const index = this.rules.findIndex(r => r.id === id);
    if (index === -1) return false;
    
    this.stopRuleMonitoring(id);
    this.rules.splice(index, 1);
    this.saveRules();
    return true;
  }
  
  // ============ 告警管理 ============
  
  getAlerts(options?: {
    status?: Alert['status'];
    type?: AlertType;
    severity?: AlertSeverity;
    limit?: number;
  }): Alert[] {
    let result = [...this.alerts];
    
    if (options?.status) {
      result = result.filter(a => a.status === options.status);
    }
    if (options?.type) {
      result = result.filter(a => a.type === options.type);
    }
    if (options?.severity) {
      result = result.filter(a => a.severity === options.severity);
    }
    
    result.sort((a, b) => b.createdAt - a.createdAt);
    
    if (options?.limit) {
      result = result.slice(0, options.limit);
    }
    
    return result;
  }
  
  getStats(): AlertStats {
    const active = this.alerts.filter(a => a.status === 'active');
    return {
      total: this.alerts.length,
      active: active.length,
      warning: active.filter(a => a.severity === 'warning').length,
      error: active.filter(a => a.severity === 'error').length,
      critical: active.filter(a => a.severity === 'critical').length
    };
  }
  
  acknowledge(id: string): boolean {
    const alert = this.alerts.find(a => a.id === id);
    if (!alert || alert.status !== 'active') return false;
    
    alert.status = 'acknowledged';
    alert.acknowledgedAt = Date.now();
    this.saveAlerts();
    this.emit('alert:acknowledged', alert);
    return true;
  }
  
  dismiss(id: string): boolean {
    const index = this.alerts.findIndex(a => a.id === id);
    if (index === -1) return false;
    
    this.alerts.splice(index, 1);
    this.saveAlerts();
    this.emit('alert:dismissed', id);
    return true;
  }
  
  dismissAll(): void {
    this.alerts = this.alerts.filter(a => a.status !== 'acknowledged');
    this.saveAlerts();
    this.emit('alert:dismiss-all');
  }
  
  // ============ 监控逻辑 ============
  
  private startMonitoring(): void {
    // 加载默认规则
    if (this.rules.length === 0) {
      this.loadDefaultRules();
    }
    
    // 启动所有规则监控
    this.rules.filter(r => r.enabled).forEach(rule => {
      this.startRuleMonitoring(rule);
    });
  }
  
  private startRuleMonitoring(rule: AlertRule): void {
    this.stopRuleMonitoring(rule.id);
    
    const timer = setInterval(async () => {
      await this.checkRule(rule);
    }, 10000); // 每10秒检查一次
    
    this.timers.set(rule.id, timer);
  }
  
  private stopRuleMonitoring(ruleId: string): void {
    const timer = this.timers.get(ruleId);
    if (timer) {
      clearInterval(timer);
      this.timers.delete(ruleId);
    }
  }
  
  private restartRuleMonitoring(rule: AlertRule): void {
    this.stopRuleMonitoring(rule.id);
    if (rule.enabled) {
      this.startRuleMonitoring(rule);
    }
  }
  
  private async checkRule(rule: AlertRule): Promise<void> {
    try {
      switch (rule.type) {
        case 'consumer_lag':
          await this.checkConsumerLag(rule);
          break;
        case 'connection_lost':
          await this.checkConnectionHealth(rule);
          break;
        case 'topic_growth':
          await this.checkTopicGrowth(rule);
          break;
      }
    } catch (error) {
      console.error(`Monitor check failed for rule ${rule.id}:`, error);
    }
  }
  
  // ============ 具体监控实现 ============
  
  private async checkConsumerLag(rule: AlertRule): Promise<void> {
    // 获取所有连接的消费组延迟
    const connections = await this.getActiveConnections();
    
    for (const conn of connections) {
      try {
        const groups = await this.getConsumerGroups(conn.id);
        
        for (const group of groups) {
          const lags = await this.getConsumerGroupLag(conn.id, group);
          
          for (const lag of lags) {
            const cacheKey = `${conn.id}:${group}:${lag.topic}:${lag.partition}`;
            const lastLag = this.lagCache.get(cacheKey) || 0;
            
            // 检查是否超过阈值
            if (lag.lag > rule.condition.threshold) {
              // 检查冷却时间
              const lastAlert = this.lastAlertTime.get(cacheKey);
              const now = Date.now();
              
              if (!lastAlert || (now - lastAlert) > rule.condition.cooldown * 1000) {
                this.createAlert({
                  ruleId: rule.id,
                  type: 'consumer_lag',
                  severity: this.calculateLagSeverity(lag.lag, rule.condition.threshold),
                  title: `消费延迟: ${group}`,
                  message: `Topic: ${lag.topic}, Partition: ${lag.partition}, Lag: ${lag.lag.toLocaleString()}`,
                  metadata: {
                    connectionId: conn.id,
                    consumerGroup: group,
                    topic: lag.topic,
                    partition: lag.partition,
                    currentLag: lag.lag,
                    threshold: rule.condition.threshold
                  }
                });
                
                this.lastAlertTime.set(cacheKey, now);
              }
            }
            
            this.lagCache.set(cacheKey, lag.lag);
          }
        }
      } catch (error) {
        console.error(`Failed to check lag for connection ${conn.id}:`, error);
      }
    }
  }
  
  private async checkConnectionHealth(rule: AlertRule): Promise<void> {
    const connections = await this.getActiveConnections();
    
    for (const conn of connections) {
      try {
        const isHealthy = await this.checkConnection(conn.id);
        
        if (!isHealthy) {
          const cacheKey = `conn:${conn.id}`;
          const lastAlert = this.lastAlertTime.get(cacheKey);
          const now = Date.now();
          
          if (!lastAlert || (now - lastAlert) > rule.condition.cooldown * 1000) {
            this.createAlert({
              ruleId: rule.id,
              type: 'connection_lost',
              severity: 'error',
              title: `连接断开: ${conn.name}`,
              message: `无法连接到 Kafka 集群，请检查网络或配置`,
              metadata: {
                connectionId: conn.id,
                connectionName: conn.name,
                retryCount: 0,
                maxRetries: 3
              }
            });
            
            this.lastAlertTime.set(cacheKey, now);
          }
        }
      } catch (error) {
        // 连接检查失败也是告警
      }
    }
  }
  
  private async checkTopicGrowth(rule: AlertRule): Promise<void> {
    // 实现 Topic 增长速度监控
    // 需要维护历史数据来计算增长率

  // ============ 告警创建 ============
  
  private createAlert(data: Omit<Alert, 'id' | 'createdAt' | 'status' | 'notified'>): void {
    const alert: Alert = {
      ...data,
      id: crypto.randomUUID(),
      createdAt: Date.now(),
      status: 'active',
      notified: {
        desktop: false,
        sound: false
      }
    };
    
    // 限制告警数量
    if (this.alerts.length >= this.MAX_ALERTS) {
      // 删除最旧的已确认告警
      const oldAckIndex = this.alerts.findIndex(a => a.status === 'acknowledged');
      if (oldAckIndex !== -1) {
        this.alerts.splice(oldAckIndex, 1);
      } else {
        // 如果没有已确认的，删除最旧的
        this.alerts.pop();
      }
    }
    
    this.alerts.unshift(alert);
    this.saveAlerts();
    
    // 发送通知
    this.sendNotification(alert);
    
    // 触发事件
    this.emit('alert:new', alert);
  }
  
  private async sendNotification(alert: Alert): Promise<void> {
    const rule = this.rules.find(r => r.id === alert.ruleId);
    if (!rule) return;
    
    // 桌面通知
    if (rule.notification.desktop && Notification.permission === 'granted') {
      try {
        new Notification(alert.title, {
          body: alert.message,
          icon: '/logo.png',
          tag: alert.id,
          requireInteraction: alert.severity === 'critical'
        });
        alert.notified.desktop = true;
      } catch (e) {
        console.error('Desktop notification failed:', e);
      }
    }
    
    // 声音通知
    if (rule.notification.sound) {
      this.playAlertSound(alert.severity);
      alert.notified.sound = true;
    }
  }
  
  private playAlertSound(severity: AlertSeverity): void {
    const audioContext = new (window.AudioContext || (window as any).webkitAudioContext)();
    const oscillator = audioContext.createOscillator();
    const gainNode = audioContext.createGain();
    
    oscillator.connect(gainNode);
    gainNode.connect(audioContext.destination);
    
    // 根据级别使用不同频率
    const frequency = {
      info: 440,
      warning: 554,
      error: 698,
      critical: 880
    }[severity];
    
    oscillator.frequency.value = frequency;
    gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);
    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.5);
    
    oscillator.start(audioContext.currentTime);
    oscillator.stop(audioContext.currentTime + 0.5);
  }
  
  private calculateLagSeverity(lag: number, threshold: number): AlertSeverity {
    const ratio = lag / threshold;
    if (ratio >= 10) return 'critical';
    if (ratio >= 5) return 'error';
    if (ratio >= 2) return 'warning';
    return 'info';
  }
  
  // ============ 存储 ============
  
  private saveAlerts(): void {
    localStorage.setItem(this.STORAGE_KEY, JSON.stringify(this.alerts));
  }
  
  private saveRules(): void {
    localStorage.setItem(this.RULES_KEY, JSON.stringify(this.rules));
  }
  
  private loadFromStorage(): void {
    try {
      const alertsData = localStorage.getItem(this.STORAGE_KEY);
      if (alertsData) {
        this.alerts = JSON.parse(alertsData);
      }
      
      const rulesData = localStorage.getItem(this.RULES_KEY);
      if (rulesData) {
        this.rules = JSON.parse(rulesData);
      }
    } catch (e) {
      console.error('Failed to load monitor data:', e);
    }
  }
  
  private loadDefaultRules(): void {
    const defaults: Omit<AlertRule, 'id'>[] = [
      {
        name: '消费延迟警告',
        enabled: true,
        type: 'consumer_lag',
        condition: {
          metric: 'lag',
          operator: 'gt',
          threshold: 10000,
          duration: 60,
          cooldown: 300
        },
        notification: {
          desktop: true,
          sound: true,
          badge: true,
          autoDismiss: 0
        }
      },
      {
        name: '连接断开告警',
        enabled: true,
        type: 'connection_lost',
        condition: {
          metric: 'connection_status',
          operator: 'eq',
          threshold: 0,
          duration: 0,
          cooldown: 60
        },
        notification: {
          desktop: true,
          sound: true,
          badge: true,
          autoDismiss: 0
        }
      }
    ];
    
    defaults.forEach(rule => this.addRule(rule));
  }
  
  // ============ 辅助方法（需接入现有服务） ============
  
  private async getActiveConnections(): Promise<Array<{id: string; name: string}>> {

    return [];
  }
  
  private async getConsumerGroups(connectionId: string): Promise<string[]> {

    lag: number;
  }>> {
    // 接入 tauriService
    return [];
  }
  
  private async checkConnection(connectionId: string): Promise<boolean> {
    // 接入 tauriService

  }
  
  // 清理资源
  destroy(): void {
    this.timers.forEach(timer => clearInterval(timer));
    this.timers.clear();
    this.removeAllListeners();
  }
}

export default MonitorService.getInstance();
```

```typescript
  );
}
```

### 2.1 当前问题分析

```typescript

```

| 问题 | 影响 | 解决方案 |
|------|------|----------|
| 全量渲染 | DOM 节点过多 | react-window 虚拟滚动 |
| 无高度回收 | 内存占用高 | 固定行高 + 缓存 |
| 展开状态混乱 | 滚动后状态丢失 | 独立状态管理 |
| 自动滚动冲突 | 虚拟滚动位置计算 | 手动控制滚动位置 |

```typescript
// components/VirtualMessageList/index.tsx

import { VariableSizeList as List } from 'react-window';
import AutoSizer from 'react-virtualized-auto-sizer';
import { useRef, useCallback, useState, useEffect, memo } from 'react';

interface VirtualMessageListProps {
  messages: KafkaMessage[];
  filteredMessages: KafkaMessage[];
  isConsuming: boolean;
  autoScroll: boolean;
  onScrollStateChange: (isNearBottom: boolean) => void;
}

// 默认行高
const DEFAULT_ROW_HEIGHT = 40;
// 展开后的额外高度
const EXPANDED_EXTRA_HEIGHT = 300;

export const VirtualMessageList = memo(({
  messages,
  filteredMessages,
  isConsuming,
  autoScroll,
  onScrollStateChange
}: VirtualMessageListProps) => {
  const listRef = useRef<List>(null);
  const rowHeightsRef = useRef<Map<number, number>>(new Map());
  const [expandedRows, setExpandedRows] = useState<Set<string>>(new Set());
  const [isAutoScrolling, setIsAutoScrolling] = useState(false);
  
  // 行高计算
  const getItemSize = useCallback((index: number) => {
    const msg = filteredMessages[index];
    const id = `${msg.partition}-${msg.offset}`;
    
    if (expandedRows.has(id)) {
      return DEFAULT_ROW_HEIGHT + EXPANDED_EXTRA_HEIGHT;
    }
    
    return rowHeightsRef.current.get(index) || DEFAULT_ROW_HEIGHT;
  }, [filteredMessages, expandedRows]);
  
  // 重置行高缓存（当过滤条件改变时）
  const resetHeightCache = useCallback(() => {
    rowHeightsRef.current.clear();
    listRef.current?.resetAfterIndex(0);
  }, []);
  
  useEffect(() => {
    resetHeightCache();
  }, [filteredMessages, resetHeightCache]);
  
  // 自动滚动处理
  useEffect(() => {
    if (autoScroll && filteredMessages.length > 0 && !isAutoScrolling) {
      setIsAutoScrolling(true);
      listRef.current?.scrollToItem(filteredMessages.length - 1, 'end');
      
      setTimeout(() => {
        setIsAutoScrolling(false);
      }, 100);
    }
  }, [filteredMessages.length, autoScroll, isAutoScrolling]);
  
  // 展开/折叠处理
  const toggleExpand = useCallback((msg: KafkaMessage) => {
    const id = `${msg.partition}-${msg.offset}`;
    
    setExpandedRows(prev => {
      const next = new Set(prev);
      const index = filteredMessages.findIndex(
        m => m.partition === msg.partition && m.offset === msg.offset
      );
      
      if (next.has(id)) {
        next.delete(id);
      } else {
        next.add(id);
      }
      
      // 通知列表更新高度
      if (index !== -1) {
        listRef.current?.resetAfterIndex(index);
      }
      
      return next;
    });
  }, [filteredMessages]);
  
  // 滚动处理
  const handleScroll = useCallback(({ scrollOffset, scrollDirection }: {
    scrollOffset: number;
    scrollDirection: 'forward' | 'backward';
  }) => {
    if (!listRef.current) return;
    
    const { scrollHeight, clientHeight } = listRef.current;
    const isNearBottom = scrollHeight - scrollOffset - clientHeight < 100;
    
    onScrollStateChange(isNearBottom);
  }, [onScrollStateChange]);
  
  // 行渲染
  const Row = useCallback(({ index, style }: { index: number; style: React.CSSProperties }) => {
    const msg = filteredMessages[index];
    const id = `${msg.partition}-${msg.offset}`;
    const isExpanded = expandedRows.has(id);
    
    return (
      <div style={style} className="border-b border-border">
        <MessageRow
          msg={msg}
          index={messages.indexOf(msg)}
          isExpanded={isExpanded}
          onToggleExpand={() => toggleExpand(msg)}
        />
      </div>
    );
  }, [filteredMessages, messages, expandedRows, toggleExpand]);
  
  if (messages.length === 0) {
    return (
      <div className="flex-1 flex items-center justify-center text-muted-foreground">
        {isConsuming ? '等待消息...' : '点击"开始"按钮消费消息'}
      </div>
    );
  }
  
  if (filteredMessages.length === 0) {
    return (
      <div className="flex-1 flex items-center justify-center text-muted-foreground">
        无搜索结果
      </div>
    );
  }
  
  return (
    <div className="flex-1 min-h-0 relative">
      <AutoSizer>
        {({ height, width }) => (
          <List
            ref={listRef}
            height={height}
            itemCount={filteredMessages.length}
            itemSize={getItemSize}
            width={width}
            onScroll={handleScroll}
            overscanCount={5} // 预渲染5行
          >
            {Row}
          </List>
        )}
      </AutoSizer>
      
      {/* 消息计数器 */}
      <div className="absolute bottom-2 right-2 bg-background/90 border border-border rounded px-2 py-1 text-xs text-muted-foreground shadow z-10">
        {filteredMessages.length} / {messages.length}
      </div>
    </div>
  );
});

}
```
