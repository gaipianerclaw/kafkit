## 一、核心功能（P0）

### 1.1 生产者页面 - 脚本模式 🎯

**需求背景**: 在生产者页面新增"脚本模式"，用户可以通过编写 JS/Python 脚本动态生成要发送的消息，支持复杂的数据构造、批量生成和定时发送。

#### 1.1.1 脚本上下文 API

```typescript
// services/scriptEngine/types.ts

/**
 * 脚本执行上下文
 * 在脚本中通过 `ctx` 或 `context` 访问
 */
interface ScriptContext {
  // ===== 消息信息（只读）=====
  message: {
    index: number;           // 当前消息序号（从1开始）
    batchIndex: number;      // 当前批次序号
    batchSize: number;       // 批次大小
    totalPlanned: number;    // 计划发送总数
  };
  
  // ===== 上次消息（用于保持状态）=====
  lastMessage?: any;         // 上次生成的消息
  
  // ===== 自定义状态存储 =====
  state: Record<string, any>;
  
  // ===== 内置工具函数 =====
  utils: {
    // 随机数生成
    randomInt(min: number, max: number): number;
    randomFloat(min: number, max: number): number;
    randomBoolean(): boolean;
    randomString(length: number, charset?: string): string;
    randomUUID(): string;
    randomChoice<T>(array: T[]): T;
    randomDate(start?: Date, end?: Date): Date;
    
    // 时间相关
    timestamp(): number;
    isoTimestamp(): string;
    formatDate(date: Date, format: string): string;
    now(): Date;
    
    // 编码/加密
    base64Encode(str: string): string;
    base64Decode(str: string): string;
    md5(str: string): string;
    sha1(str: string): string;
    sha256(str: string): string;
    
    // 数据生成
    lorem(words: number): string;
    name(): string;
    email(): string;
    phone(): string;
    address(): string;
    company(): string;
    
    // 数值递增
    sequence(start?: number, step?: number): number;
    
    // 数组操作
    range(start: number, end: number, step?: number): number[];
    shuffle<T>(array: T[]): T[];
    sample<T>(array: T[], count: number): T[];
  };
}
```

```javascript
function generateMessage(ctx) {
  const products = ['iPhone', 'iPad', 'MacBook', 'AirPods'];
  const states = ['pending', 'paid', 'shipped', 'delivered'];
  
  return {
    orderId: `ORD-${ctx.utils.randomString(8).toUpperCase()}`,
    userId: `USER-${ctx.utils.randomInt(1000, 9999)}`,
    product: ctx.utils.randomChoice(products),
    quantity: ctx.utils.randomInt(1, 5),
    price: ctx.utils.randomFloat(99.99, 9999.99),
    state: ctx.utils.randomChoice(states),

  };
}
generateMessage(context);
```

```javascript
// 高级示例 - 状态保持（模拟传感器连续数据）
function generateMessage(ctx) {
  // 获取上次温度，保持连续性
  let lastTemp = ctx.state.lastTemp || 25;
  
  // 温度小幅随机波动
  const change = ctx.utils.randomFloat(-0.5, 0.5);
  const newTemp = Math.max(15, Math.min(35, lastTemp + change));
  
  // 保存状态供下次使用
  ctx.state.lastTemp = newTemp;
  
  return {
    deviceId: `DEV-${ctx.utils.randomString(6)}`,
    temperature: parseFloat(newTemp.toFixed(2)),
    humidity: ctx.utils.randomInt(30, 80),
    timestamp: ctx.utils.timestamp(),
    alert: newTemp > 30 || newTemp < 18
  };
}
generateMessage(context);
```

#### 1.2.2 JavaScript 引擎实现

```typescript
// services/scriptEngine/JSEngine.ts
import { ScriptEngine, ScriptContext, ValidationResult } from './types';

export class JSEngine implements ScriptEngine {
  private quickJS: any; // QuickJS 实例
  
  async initialize(): Promise<void> {
    // 初始化 QuickJS WASM
    const QuickJS = await import('quickjs-emscripten');
    this.quickJS = await QuickJS.getQuickJS();
  }
  
  async execute(script: string, context: ScriptContext): Promise<any> {
    // 创建新的运行时（隔离环境）
    const runtime = this.quickJS.newRuntime();
    runtime.setMemoryLimit(1024 * 1024); // 1MB 内存限制
    runtime.setMaxStackSize(1024 * 1024); // 1MB 栈限制
    
    const vm = runtime.newContext();
    
    // 注入上下文
    const contextHandle = vm.newObject();
    this.injectContext(vm, contextHandle, context);
    vm.setProp(vm.global, 'context', contextHandle);
    vm.setProp(vm.global, 'ctx', contextHandle);
    
    // 执行脚本
    const result = vm.evalCode(script, {
      maxSteps: 100000, // 执行步数限制
      timeout: 100, // 100ms 超时
    });
    
    // 处理结果
    if (result.error) {
      const error = vm.dump(result.error);
      result.error.dispose();
      throw new Error(error);
    }
    
    const value = vm.dump(result.value);
    result.value.dispose();
    
    // 清理
    vm.dispose();
    runtime.dispose();
    
    return value;
  }
  
  async validate(script: string): Promise<ValidationResult> {
    try {
      const runtime = this.quickJS.newRuntime();
      const vm = runtime.newContext();
      
      // 只进行语法检查，不执行
      const result = vm.evalCode(`(() => { ${script}; return 'ok'; })()`);
      
      const isValid = !result.error;
      const error = result.error ? vm.dump(result.error) : undefined;
      
      if (result.error) result.error.dispose();
      if (result.value) result.value.dispose();
      vm.dispose();
      runtime.dispose();
      
      return { valid: isValid, error };
    } catch (e) {
      return { valid: false, error: e.message };
    }
  }
  
  private injectContext(vm: any, handle: any, context: ScriptContext): void {
    // 注入 message
    const msgHandle = vm.newObject();
    vm.setProp(msgHandle, 'index', vm.newNumber(context.message.index));
    vm.setProp(msgHandle, 'batchIndex', vm.newNumber(context.message.batchIndex));
    vm.setProp(msgHandle, 'batchSize', vm.newNumber(context.message.batchSize));
    vm.setProp(msgHandle, 'totalPlanned', vm.newNumber(context.message.totalPlanned));
    vm.setProp(handle, 'message', msgHandle);
    
    // 注入 utils
    const utilsHandle = vm.newObject();
    this.injectUtils(vm, utilsHandle);
    vm.setProp(handle, 'utils', utilsHandle);
    
    // 注入 state
    const stateHandle = vm.newObject();
    Object.entries(context.state).forEach(([key, value]) => {
      vm.setProp(stateHandle, key, this.toVMValue(vm, value));
    });
    vm.setProp(handle, 'state', stateHandle);
    
    // 注入 lastMessage（如果存在）
    if (context.lastMessage) {
      vm.setProp(handle, 'lastMessage', this.toVMValue(vm, context.lastMessage));
    }
  }
  
  private injectUtils(vm: any, handle: any): void {
    // 随机数
    this.defineFunction(vm, handle, 'randomInt', (min, max) => 
      Math.floor(Math.random() * (max - min + 1)) + min);
    this.defineFunction(vm, handle, 'randomFloat', (min, max) => 
      Math.random() * (max - min) + min);
    this.defineFunction(vm, handle, 'randomUUID', () => 
      crypto.randomUUID());
    // ... 其他工具函数
  }
  
  private defineFunction(vm: any, handle: any, name: string, fn: Function): void {
    const fnHandle = vm.newFunction(name, (...args: any[]) => {
      const result = fn(...args.map((a: any) => vm.dump(a)));
      return this.toVMValue(vm, result);
    });
    vm.setProp(handle, name, fnHandle);
    fnHandle.dispose();
  }
  
  private toVMValue(vm: any, value: any): any {
    if (value === null || value === undefined) return vm.undefined;
    if (typeof value === 'boolean') return vm.newBoolean(value);
    if (typeof value === 'number') return vm.newNumber(value);
    if (typeof value === 'string') return vm.newString(value);
    if (typeof value === 'object') {
      const obj = vm.newObject();
      Object.entries(value).forEach(([k, v]) => {
        vm.setProp(obj, k, this.toVMValue(vm, v));
      });
      return obj;
    }
    return vm.undefined;
  }
}
```

### 2.1 数据导入（P1）

在脚本模式基础上，支持从文件导入消息模板：

**支持格式**:
- JSON Array / JSON Lines
- CSV（自动转 JSON）
- Excel（多 Sheet）
- Avro（二进制）

**工作流程**:
1. 上传文件
2. 预览前 10 条数据
3. 字段映射（CSV/Excel）
4. 自动生成脚本或作为数据源
5. 批量发送

- 模板分类和搜索

```javascript
function generateMessage(ctx) {
  const products = [
    { name: 'iPhone 15', price: 5999 },
    { name: 'iPad Pro', price: 6999 },
    { name: 'MacBook Air', price: 8999 }
  ];
  
  const product = ctx.utils.randomChoice(products);
  
  return {
    eventType: 'ORDER_CREATED',
    orderId: `ORD-${ctx.utils.randomString(10).toUpperCase()}`,
    userId: `USER-${ctx.utils.randomInt(10000, 99999)}`,
    product: product.name,
    price: product.price,
    quantity: ctx.utils.randomInt(1, 3),
    status: 'pending',
    createdAt: ctx.utils.isoTimestamp(),
    metadata: {
      source: 'web',
      version: '2.0'
    }
  };
}

generateMessage(context);
```

### 示例 2: 模拟 IoT 传感器
```javascript
function generateMessage(ctx) {
  // 获取上次状态，保持数据连续性
  const lastTemp = ctx.state.lastTemp || 25;
  const lastHumidity = ctx.state.lastHumidity || 50;
  
  // 模拟环境变化
  const tempChange = ctx.utils.randomFloat(-0.3, 0.3);
  const humidityChange = ctx.utils.randomFloat(-1, 1);
  
  const newTemp = Math.max(15, Math.min(35, lastTemp + tempChange));
  const newHumidity = Math.max(20, Math.min(90, lastHumidity + humidityChange));
  
  // 保存状态
  ctx.state.lastTemp = newTemp;
  ctx.state.lastHumidity = newHumidity;
  
  return {
    deviceId: `DEV-${ctx.utils.randomString(8)}`,
    sensorType: 'DHT22',
    temperature: parseFloat(newTemp.toFixed(2)),
    humidity: parseFloat(newHumidity.toFixed(1)),
    timestamp: ctx.utils.timestamp(),
    alerts: {
      highTemp: newTemp > 30,
      lowHumidity: newHumidity < 30
    }
  };
}

generateMessage(context);
```

### 示例 3: 模拟股票行情
```javascript
function generateMessage(ctx) {
  const symbols = ['AAPL', 'GOOGL', 'MSFT', 'AMZN', 'TSLA'];
  const symbol = ctx.utils.randomChoice(symbols);
  
  // 获取上次价格（模拟股价连续性）
  const stateKey = `price_${symbol}`;
  const basePrice = ctx.state[stateKey] || ctx.utils.randomFloat(100, 500);
  
  // 模拟价格波动 (-2% ~ +2%)
  const changePercent = ctx.utils.randomFloat(-0.02, 0.02);
  const newPrice = basePrice * (1 + changePercent);
  
  // 保存新价格
  ctx.state[stateKey] = newPrice;
  
  const volume = ctx.utils.randomInt(1000, 100000);
  
  return {
    symbol: symbol,
    price: parseFloat(newPrice.toFixed(2)),
    change: parseFloat((newPrice - basePrice).toFixed(2)),
    changePercent: parseFloat((changePercent * 100).toFixed(2)),
    volume: volume,
    timestamp: ctx.utils.isoTimestamp(),
    market: 'NASDAQ'
  };
}

generateMessage(context);
```
